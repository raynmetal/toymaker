263 inline std::pair<glm::vec3, glm::vec3> deriveTangents(const glm::vec3& vector) {
264 assert(isFinite(vector) && "Vector must be finite");
265 assert(squareDistance(vector) && "Zero vectors are invalid as basis for tangents");
266
267 const auto normalized { glm::normalize(vector) };
268 const glm::vec3 tangentPrelim { ((glm::abs(normalized.x) >= .57735f) ?
269 glm::vec3{ normalized.y, -normalized.x, 0.f } : glm::vec3{ 0.f, normalized.z, -normalized.y }
270 ) };
271 const glm::vec3 tangent2 {
272 glm::normalize(glm::cross(tangentPrelim, normalized))
273 };
274
275 return { glm::normalize(glm::cross(normalized, tangent2)), tangent2 };
276 }

312 template <typename T, typename U>
313 inline std::pair<bool, Simplex> gjkOverlaps(const T& one, const U& two) {
314 assert(one.isSensible() && two.isSensible() && "Invalid object bounds provided");
315
316 // Both need to be non-degenerate bounds to actually overlap
317 if(!one.isPositiveStrict() || !two.isPositiveStrict()) {
318 return { false, Simplex {} };
319 }
320
321 glm::vec3 searchDirection { two.getPositionWorld() - one.getPositionWorld() };
322
323 // Two non-degenerate objects share the same origin, obviously they overlap
324 if(squareDistance(searchDirection) == 0.f) {
325 return { true, Simplex {} };
326 }
327
328 Simplex simplex {};
329 auto fullDifference { minkowskiDifferenceFull(one, two, searchDirection) };
330 const auto& candidatePoint { std::get<0>(fullDifference) }; // updated with fullDifference
331 const auto& supportA { std::get<1>(fullDifference) }; // updated with fullDifference
332 const auto& supportB { std::get<2>(fullDifference) }; // updated with fullDifference
333 simplex.append(candidatePoint, supportA, supportB);
334
335 // we go towards the origin from the first point we found
336 glm::vec3 toOrigin { -simplex.mPoints[0] };
337 searchDirection = toOrigin;
338 for(uint8_t iteration { 0 }; squareDistance(toOrigin) && iteration < kMaxIterations; ++iteration) {
339 // find a new search direction
340 searchDirection = toOrigin;
341
342 // find a new candidate point
343 fullDifference = minkowskiDifferenceFull(one, two, searchDirection);
344
345 // we couldn't find a point far enough past the origin in this direction, so
346 // obviously there is no intersection
347 if(glm::dot(candidatePoint, searchDirection) <= 0.f) {
348 return { false, simplex };
349 }
350
351 simplex.append(candidatePoint, supportA, supportB);
352 const auto result { simplex.evaluate() };
353 toOrigin = result.second;
354 if(result.first) {
355 return { true, simplex };
356 }
357 }
358
359 // handle degenerate simplexes by forcing them to expand to a full simplex
360 while(simplex.mNPoints < 4) {
361 switch(simplex.mNPoints) {
362 case 1:
363 simplex.mPoints[simplex.mNPoints] = glm::vec3 { 2.f * kSimplexEpsilon, 0.f, 0.f };
364 simplex.mPointsSupportA[simplex.mNPoints] = glm::vec3 { kSimplexEpsilon, 0.f, 0.f };
365 simplex.mPointsSupportB[simplex.mNPoints] = -glm::vec3 { kSimplexEpsilon, 0.f, 0.f };
366 ++simplex.mNPoints;
367 break;
368 case 2:
369 {
370 const glm::vec3 dirAB { glm::normalize(simplex.mPoints[1] - simplex.mPoints[0]) };
371 assert(squareDistance(dirAB) > 0 && "A and B should never be the same point");
372 const glm::vec3 dirOffset { (dirAB.x != 0.f || dirAB.z != 0.f)?
373 glm::vec3{ 0.f, 1.f, 0.f } : glm::vec3{ 1.f, 0.f, 1.f }
374 };
375 searchDirection = glm::normalize(-dirAB + dirOffset);
376 simplex.mPoints[simplex.mNPoints] = 2.f * kSimplexEpsilon * dirOffset;
377 simplex.mPointsSupportA[simplex.mNPoints] = kSimplexEpsilon * dirOffset;
378 simplex.mPointsSupportB[simplex.mNPoints] = -kSimplexEpsilon * dirOffset;
379 ++simplex.mNPoints;
380 }
381 break;
382 case 3:
383 {
384 const glm::vec3 normABC { glm::normalize(glm::cross(simplex.mPoints[1] - simplex.mPoints[0], simplex.mPoints[2] - simplex.mPoints[0])) };
385 assert(isNumber(normABC) && isFinite(normABC) && "Invalid norm computed, degenerate triangle detected.");
386 simplex.mPoints[simplex.mNPoints] = 2.f * kSimplexEpsilon * normABC;
387 simplex.mPointsSupportA[simplex.mNPoints] = kSimplexEpsilon * normABC;
388 simplex.mPointsSupportB[simplex.mNPoints] = -kSimplexEpsilon * normABC;
389 ++simplex.mNPoints;
390 }
391 break;
392 default:
393 assert(false && "Simplex with invalid number of points for this section of the algorithm provided");
394 break;
395 };
396 }
397
398 return { true, simplex };
399 }

402 template <typename T, typename U>
403 inline Collision checkCollision(const T& one, const U& two) {
404 // perform inexpensive AABB test first
405 if(!overlaps(AxisAlignedBounds(one), AxisAlignedBounds(two))) {
406 return {
407 .mCollided { false }
408 };
409 }
410
411 // special case: sphere sphere collision, solvable in constant time
412 if(
413 one.mType == ObjectBounds::TrueVolumeType::SPHERE
414 && two.mType == ObjectBounds::TrueVolumeType::SPHERE
415 ) {
416 return checkCollisionSphereSphere(one, two);
417 }
418
419 // check via GJK whether there's any overlap at all
420 const std::pair<bool, Simplex> overlapResult { gjkOverlaps(one, two) };
421 if(!overlapResult.first) {
422 return {
423 .mCollided { false },
424 };
425 }
426
427
428 // allocate result storage
429 Collision collisionResult { Collision {
430 .mCollided { true },
431 } };
432
433 // build a polytope containing the closest surface to the origin in
434 // the Minkowski difference between the objects
435 const Polytope polytope { buildPolytope(one, two, overlapResult.second) };
436
437 // derive barycentric coordinates for the contact point relative to the
438 // triangle in the difference shape it lies on
439 const glm::vec3 closestPoint { polytope.getClosestPoint() };
440 const AreaTriangle closestTriangle { polytope.getClosestTriangle() };
441 const glm::mat3 barycentricSolver { computeBarycentricSolver(closestTriangle) };
442 // If we can't solve this, then there's nothing for us to do
443 if(!isSensible(barycentricSolver)) {
444 return { false };
445 }
446
447 const glm::vec3 barycentricCoordinates {
448 barycentricSolver * closestPoint
449 };
450
451 // Use the barycentric coordinates to derive contact point for both shapes
452 const AreaTriangle closestTriangleOne {
453 polytope.getClosestTriangleSupportA()
454 };
455 const AreaTriangle closestTriangleTwo {
456 polytope.getClosestTriangleSupportB()
457 };
458 const glm::mat3 barycentricOne {
459 closestTriangleOne.mPoints[0], closestTriangleOne.mPoints[1], closestTriangleOne.mPoints[2]
460 };
461 const glm::mat3 barycentricTwo {
462 closestTriangleTwo.mPoints[0], closestTriangleTwo.mPoints[1], closestTriangleTwo.mPoints[2]
463 };
464 collisionResult.mContactA.mPoint = barycentricOne * barycentricCoordinates;
465 assert(
466 isNumber(collisionResult.mContactA.mPoint) && isFinite(collisionResult.mContactA.mPoint)
467 && "Invalid contact point computed for object A"
468 );
469 collisionResult.mContactB.mPoint = barycentricTwo * barycentricCoordinates;
470 assert(
471 isNumber(collisionResult.mContactB.mPoint) && isFinite(collisionResult.mContactB.mPoint)
472 && "Invalid contact point computed for object B"
473 );
474
475 // build tangent vectors to the normal (used for friction calculations)
476 const auto contactNormal {
477 squareDistance(closestPoint)? closestPoint: one.getPositionWorld() - two.getPositionWorld()
478 };
479 const auto tangentPair { deriveTangents(contactNormal) };
480 collisionResult.mContactB.mNormal = glm::normalize(contactNormal);
481 collisionResult.mContactB.mTangent1 = glm::normalize(tangentPair.first);
482 collisionResult.mContactB.mTangent2 = glm::normalize(tangentPair.second);
483 collisionResult.mContactA.mNormal = -collisionResult.mContactB.mNormal;
484 collisionResult.mContactA.mTangent1 = -collisionResult.mContactB.mTangent1;
485 collisionResult.mContactA.mTangent2 = -collisionResult.mContactB.mTangent2;
486
487 // set penetration depth value for both contacts
488 collisionResult.mContactA.mPenetrationDepth
489 = collisionResult.mContactB.mPenetrationDepth
490 = glm::length(closestPoint);
491
492 return collisionResult;
493 }

495 template <typename T, typename U>
496 inline Polytope buildPolytope(const T& one, const U& two, const Simplex& simplex) {
497 Polytope polytope { simplex };
498 glm::vec3 searchDirection { polytope.getNextSearch() };
499 assert(squareDistance(searchDirection) > 0 && "Search direction should always be non-zero");
500 assert(isFinite(searchDirection) && "Search direction should always be finite");
501
502 // Compute first support point
503 std::tuple<glm::vec3, glm::vec3, glm::vec3> differenceFull { minkowskiDifferenceFull(one, two, searchDirection) };
504
505 // These reference types can be thought of as "views" into the
506 // differenceFull struct
507 const glm::vec3& supportA { std::get<1>(differenceFull) };
508 const glm::vec3& supportB { std::get<2>(differenceFull) };
509 const glm::vec3& difference { std::get<0>(differenceFull) };
510
511 // Keep building out the expanding polytope until either the minimum translation vector is found or we run out of iterations
512 uint8_t iterationsLeft { kMaxIterations };
513 while(polytope.append(difference, supportA, supportB) && iterationsLeft--) {
514 searchDirection = polytope.getNextSearch();
515 differenceFull = minkowskiDifferenceFull(one, two, searchDirection);
516 }
517
518 return polytope;
519 }
520}
