428 template<typename TConstraint, typename TConstraintData>
429 PhysicsSystem::ConstraintID PhysicsSystem::registerConstraint(const std::vector<std::pair<EntityID, TConstraintData>>& participants, float compliance) {
430 // reclaim a deleted constraint id if possible
431 ConstraintID newConstraint { mConstraints.size() };
432 if(!mConstraintsDeleted.empty()) {
433 newConstraint = *mConstraintsDeleted.begin();
434 mConstraintsDeleted.erase(newConstraint);
435 }
436
437 // get separate constraint parameter and participating entity lists
438 std::vector<EntityID> entities {};
439 std::vector<TConstraintData> constraintData {};
440 for(const std::pair<EntityID, TConstraintData>& participant: participants) {
441 constraintData.push_back(participant.second);
442 entities.push_back(participant.first);
443 }
444
445 // create constraint
446 if(newConstraint == mConstraints.size()) {
447 mConstraints.push_back({
448 std::make_unique<TConstraint>(constraintData, compliance),
449 entities
450 });
451 } else {
452 mConstraints[newConstraint] = {
453 std::make_unique<TConstraint>(constraintData, compliance),
454 entities
455 };
456 }
457
458 for(const auto& entity: entities) {
459 mEntityConstraintMap[entity].insert(newConstraint);
460 }
461
462 return newConstraint;
463 }
464}
