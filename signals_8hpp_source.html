469 template <typename ...TArgs>
470 void Signal_<TArgs...>::emit (TArgs ... args) {
471 //observers that will be removed from the list after this signal has been emitted
472 std::vector<std::weak_ptr<SignalObserver_<TArgs...>>> expiredObservers {};
473
474 for(auto observer: mObservers) {
475 // lock means that this observer is still active
476 if(std::shared_ptr<SignalObserver_<TArgs...>> activeObserver = observer.lock()) {
477 (*activeObserver)(args...);
478
479 // go to the purge list
480 } else {
481 expiredObservers.push_back(observer);
482 }
483 }
484
485 // remove dead observers
486 for(auto expiredObserver: expiredObservers) {
487 mObservers.erase(expiredObserver);
488 }
489 }
